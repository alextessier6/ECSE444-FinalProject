int fgetc(FILE *f) {
  uint8_t ch = 0;
  while (HAL_OK != HAL_UART_Receive(&handle, (uint8_t *)&ch, 1, 30000));
  return ch;
}

/**
  * @brief System Clock Configuration
  * @retval None
  */
void SystemClock_Config(void)
{

  RCC_OscInitTypeDef RCC_OscInitStruct;
  RCC_ClkInitTypeDef RCC_ClkInitStruct;

    /**Initializes the CPU, AHB and APB busses clocks 
    */
  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_MSI;
  RCC_OscInitStruct.MSIState = RCC_MSI_ON;
  RCC_OscInitStruct.MSICalibrationValue = 0;
  RCC_OscInitStruct.MSIClockRange = RCC_MSIRANGE_6;
  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_MSI;
  RCC_OscInitStruct.PLL.PLLM = 1;
  RCC_OscInitStruct.PLL.PLLN = 40;
  RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV7;
  RCC_OscInitStruct.PLL.PLLQ = RCC_PLLQ_DIV2;
  RCC_OscInitStruct.PLL.PLLR = RCC_PLLR_DIV2;
  if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
  {
    _Error_Handler(__FILE__, __LINE__);
  }

    /**Initializes the CPU, AHB and APB busses clocks 
    */
  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK|RCC_CLOCKTYPE_SYSCLK
                              |RCC_CLOCKTYPE_PCLK1|RCC_CLOCKTYPE_PCLK2;
  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV1;
  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;

  if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_4) != HAL_OK)
  {
    _Error_Handler(__FILE__, __LINE__);

    _Error_Handler(__FILE__, __LINE__);

  HAL_SYSTICK_CLKSourceConfig(SYSTICK_CLKSOURCE_HCLK);

  HAL_NVIC_SetPriority(SysTick_IRQn, 15, 0);

/* StartDefaultTask function */
void StartDefaultTask(void const * argument)
{

  /* USER CODE BEGIN 5 */
	UART_init();
	BSP_ACCELERO_Init();
	int16_t XYZ[3];
  /* Infinite loop */
  for(;;)
  {
		osDelay(100);
		BSP_ACCELERO_AccGetXYZ(XYZ);
		printf("X = %d, Y = %d, Z = %d\n", XYZ[0], XYZ[1], XYZ[2]);
  }
  /* USER CODE END 5 */ 
}

void StartSensor(void const * argument)
{
	BSP_ACCELERO_Init();
	BSP_GYRO_Init();
	BSP_HSENSOR_Init();
	BSP_MAGNETO_Init();
	BSP_PSENSOR_Init();
	while(1){
		osDelay(200); //delay 200 ms-> 5Hz
		switch(sensorIndex){
			case 0:
				//BSP_ACCELERO_Init();
				BSP_ACCELERO_AccGetXYZ(sensorOutputInt); // int *3
				//BSP_ACCELERO_DeInit();
			break;
			case 1:
				// BSP_GYRO_Init();
				BSP_GYRO_GetXYZ(sensorOutputFloat); // should be float * 3
				//BSP_GYRO_DeInit();
			break;
			case 2:
				//BSP_HSENSOR_Init();
				sensorOutputFloat[0]=BSP_HSENSOR_ReadHumidity(); // float 
				//BSP_HSENSOR_DeInit(); // no deinit 
			break;
			case 3:
				// BSP_MAGNETO_Init();
				BSP_MAGNETO_GetXYZ(sensorOutputInt);// int * 3
				// BSP_MAGNETO_DeInit();
			break;
			case 4:
				// BSP_PSENSOR_Init();
				sensorOutputFloat[0] = BSP_PSENSOR_ReadPressure(); // float 
				// BSP_PSENSOR_DeInit(); // no deinit
			break;
			case 5:
				BSP_TSENSOR_Init();
				sensorOutputFloat[0] = BSP_TSENSOR_ReadTemp(); // float
				// BSP_TSENSOR_DeInit(); // no deinit
			break;
			default:
				printf("Wrong Sensor Index: %d\n", sensorIndex);
		}
	}
}

void buttonPress(void const * argument)
{
	if (new_putton_press){
		new_putton_press = 0;
		// short press switch between sensors
		sensorIndex += 1;
		sensorIndex = sensorIndex % nSensors;
	}
	// long press wake up/sleep 
}

void printUsart(void const * argument)
{
	UART_init();
	  /* Infinite loop */
  for(;;)
  {
		osDelay(100);
		switch(sensorIndex){
			case 0: //acc
				printf("Acc X = %d, Y = %d, Z = %d\n",  sensorOutputInt[0], sensorOutputInt[1], sensorOutputInt[2]);
			break;
			case 1: //gyro 
				printf("Gyro X = %f, Y = %f, Z = %f\n", sensorOutputFloat[0], sensorOutputFloat[1], sensorOutputFloat[2]);
			break;
			case 2: // humidity 
				printf("humidity value = %f\n", sensorOutputFloat[0]);
			break;
			case 3: //mag
				printf("Mag X = %d, Y = %d, Z = %d\n",  sensorOutputInt[0], sensorOutputInt[1], sensorOutputInt[2]);
			break;
			case 4: // pressure 
				printf("pressure value = %f\n", sensorOutputFloat[0]);
			break;
			case 5: // temperature
				printf("temp value = %f\n", sensorOutputFloat[0]);
			break;
			default:
			printf("Wrong Sensor Index: %d\n", sensorIndex);
		}
		sensorIndex += 1;
   sensorIndex = sensorIndex % nSensors;
  }
  /* USER CODE END 5 */ 
}
